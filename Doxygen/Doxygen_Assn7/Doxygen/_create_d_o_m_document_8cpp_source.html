76 void CreateDOMDocument::ParseCommandString(std::string strUserEntry) {
77  // regular expressions for basic commands

79  regex reBasicPrintCommand("\\s*print.*", regex::icase);
80  regex reBasicHelpCommand("\\s*help.*", regex::icase);

93  cout << " add content element_name content_text" << endl;
