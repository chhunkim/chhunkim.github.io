335 void XMLElement::getAttribute(string strLine) {
336  // initialized strKey and strValue to be null
337  string strKey, strValue = "";
338  // initialized findOpeningTag as a position when found "<" in the strLine
339  size_t findOpeningTag = strLine.find("<");
340  // initialized findClosingTag as a position when found ">" in the strLine
341  size_t findClosingTag = strLine.find(">");
342  // initialized findSpace as a position when found " " in the strLine
343  size_t findSpace = strLine.find(" ", findOpeningTag + 1);
344  // initialized findEqualSign as a position when found "=" in the strLine
345  size_t findEqualSign = strLine.find("=", findSpace + 1);
346  // initialized findQuot1 as a position when found the first quot in the strLine
347  size_t findQuot1 = strLine.find("\"", findEqualSign + 1);
348  // initialized findQuot2 as position when found the second quot after the first double in the strLine
349  size_t findQuot2 = strLine.find("\"", findQuot1 + 1);
350  // if the "<" and "=" and "SPACE" is found in the strLine, go to body
351  if (findOpeningTag != string::npos && findEqualSign != string::npos && findSpace != string::npos) {
352  // assign all the characters between the space and the equal sign to be strKey
353  strKey = strLine.substr(findSpace + 1, findEqualSign - findSpace - 1);
354  // assign all the characters between the double quot to be strValue
355  strValue = strLine.substr(findQuot1 + 1, findQuot2 - findQuot1 - 1);
356  // Instantiated a new attribute class
357  Attributes* newAttribute = new Attributes(strKey, strValue);
358  // stored strKey and strValue into the vector of attribute vector
359  vecAttribute.push_back(newAttribute);
360 
361  // New initializations below will be used in the case there are two or more attributes
362  // in each string line.
363  // findEqualSign is assigned to be a position of the "=" after the second double quot
364  // findSpace is assigned to be a position of the "SPACE" after the second double quot
365  // findQuot1 and findQuot2 are assigned to be positions of the opening and closing double quot
366  // after the previous second quot
367  findEqualSign = strLine.find("=", findQuot2 + 1);
368  findSpace = strLine.find(" ", findQuot2 + 1);
369  findQuot1 = strLine.find("\"", findEqualSign + 1);
370  findQuot2 = strLine.find("\"", findQuot1 + 1);
371 
372  // the while loop is used for searching more attribute
373  // if there are more than two before the closing angle bracket.
374  while (findEqualSign != string::npos && findEqualSign < findClosingTag) {
375  // assign all the characters between the space and the equal sign to be strKey
376  strKey = strLine.substr(findSpace + 1, findEqualSign - findSpace - 1);
377  // assign all the characters between the double quot to be strValue
378  strValue = strLine.substr(findQuot1 + 1, findQuot2 - findQuot1 - 1);
379  // Instantiated a new attribute class
380  Attributes* newAttribute = new Attributes(strKey, strValue);
381  // stored strKey and strValue into the vector of attribute vector
382  vecAttribute.push_back(newAttribute);
383  // reset the position for "SPACE" and "=" and the opening and closing double quot
384  findSpace = strLine.find(" ", findQuot2 + 1);
385  findEqualSign = strLine.find("=", findQuot2 + 1);
386  findQuot1 = strLine.find("\"", findEqualSign + 1);
387  findQuot2 = strLine.find("\"", findQuot1 + 1);
388  }
389  }
390  // called the ShowAttributes function to print out the stored attributes
391  ShowAttributes(vecAttribute);
392  // clear the strKey
393  strKey.clear();
394  // clear the strValue
395  strValue.clear();
396  return;
397 }
