185 ParserState GetXMLData(string strLine, string& strElementName, string& strContent,
186  int nStartPos, int& nEndPos, ParserState currentState) {
187  strContent = ""; // content of an element
188  strElementName = ""; // name of element found
189  nStartPos = // position of an opening < character
190  strLine.find("<", nStartPos);
191  nEndPos = // position of the corresponding closing > character
192  strLine.find(">", nStartPos);
193 
194  // // for debugging
195  // if ( nStartPos != string::npos ) {
196  // cout << "|" << strLine.substr( nStartPos, 4 ) << "|" << endl ;
197  // }
198 
199  // closing > not found (string::npos is returned if the search string is not found)
200  if (nEndPos == string::npos) {
201  if (currentState == STARTING_COMMENT || currentState == IN_COMMENT) {
202  strContent = strLine;
203  return IN_COMMENT;
204  } else if (strLine.substr(nStartPos, 4).compare("<!--") == 0) {
205  return STARTING_COMMENT;
206  } else {
207  strContent = strLine;
208  trim(strContent);
209  return UNKNOWN;
210  }
211 
212  // opening < not found
213  } else if (nStartPos == string::npos && nEndPos > 1 &&
214  strLine.substr(nEndPos - 2, 3).compare("-->") == 0) {
215  return ENDING_COMMENT;
216 
217  // ? found after opening <, indicating a directive
218  } else if (strLine[nStartPos + 1] == '?') {
219  strContent = strLine.substr(nStartPos + 2, (nEndPos - nStartPos - 3));
220  return DIRECTIVE;
221 
222  // !-- found after opening <, indicating a comment
223  } else if (strLine.substr(nStartPos, 4).compare("<!--") == 0) {
224  strContent = strLine.substr(nStartPos + 4, nEndPos - nStartPos - 6);
225  trim(strContent);
226  return ONE_LINE_COMMENT;
227 
228  // found element closing tag
229  } else if (strLine[nStartPos + 1] == '/') {
230  strElementName = strLine.substr(nStartPos + 2, (nEndPos - nStartPos - 2));
231 
232  // element name should match last entry pushed onto stack
233  if (strElementName.compare(vecElementNames.back()) == 0) {
234  vecElementNames.pop_back();
235  return ELEMENT_CLOSING_TAG;
236  } else {
237  strContent = "XML file is not well-formed. Read element closing tag " +
238  strElementName + ", but last element pushed onto stack was " +
239  vecElementNames.back() + ".";
240  return ERROR;
241  }
242 
243  // found element opening tag
244  } else {
245  strElementName = strLine.substr(nStartPos + 1, (nEndPos - nStartPos - 1));
246 
247  // ignore any attributes found
248  int posSpace = strElementName.find(" ", 0);
249  if (posSpace != string::npos) {
250  strElementName = strElementName.substr(0, posSpace);
251  }
252 
253  // search for closing element tag on the same line
254  int posClosingTag = strLine.find("</" + strElementName + ">", nEndPos + 1);
255 
256  // closing tag found
257  if (posClosingTag != string::npos) {
258  strContent = strLine.substr(nEndPos + 1, posClosingTag - nEndPos - 1);
259  trim(strContent);
260  return ELEMENT_NAME_AND_CONTENT;
261 
262  // search for self-closing tag
263  } else {
264  posClosingTag = strLine.find("/>", nStartPos + 1);
265 
266  if (posClosingTag != string::npos) {
267  strContent = "";
268  return ELEMENT_NAME_AND_CONTENT;
269 
270  // closing tag not found
271  } else {
272  // push element name onto vector stack
273  vecElementNames.push_back(strElementName);
274  return ELEMENT_OPENING_TAG;
275  }
276  }
277  }
278 
279  // the line could not be interpreted
280  return UNKNOWN;
281 }

288 void ParseAttributes(string strLine, Element* elem) {
289  int nStartPos; // position of an opening < character
290  int nEndPos; // position of the corresponding closing > character
291  int nSpacePos; // position of a space in the tag that delimits attributes
292  int nAnglePos; // position of a closing angle bracket in the tag that ends an attribute value
293  int nQuotePos; // position of a quote in the tag that ends an attribute value
294  int nEqualsPos; // position of an = sign
295  string key; // key of a found attribute (the part before the = sign)
296  string value; // value of a found attribute (the part after the = sign)
297 
298  // delineate the opening tag (remember that attributes can only be on opening tags)
299  nStartPos = strLine.find("<", nStartPos);
300  nEndPos = strLine.find(">", nStartPos);
301  strLine = strLine.substr(nStartPos + 1, nEndPos - nStartPos - 1);
302  trim(strLine); // always trim the string to account for double spaces
303 
304  // shorten strLine to remove the tag name
305  nSpacePos = strLine.find(" ");
306  if (nSpacePos != string::npos) {
307  strLine = strLine.substr(nSpacePos + 1, strLine.length());
308  trim(strLine); // always trim the string to account for double spaces
309  }
310 
311  // extract all attributes as key/value pairs and add them to the element's map
312  while (strLine.find("=") != string::npos) {
313  // find the = sign and extract the attribute key
314  nEqualsPos = strLine.find("=");
315  key = strLine.substr(0, nEqualsPos);
316  trim(key); // always trim the string to account for double spaces
317  // shorten strLine to remove the key we just extracted
318  strLine = strLine.substr(nEqualsPos + 1, strLine.length());
319  trim(strLine); // always trim the string to account for double spaces
320 
321  // extract the attribute value, which is terminated either by a space or a >
322  nSpacePos = strLine.find(" ");
323  if (nSpacePos != string::npos) { // terminated by a space character
324  value = strLine.substr(0, nSpacePos);
325  // shorten strLine to remove the key we just extracted
326  strLine = strLine.substr(nSpacePos + 1, strLine.length());
327  trim(strLine); // always trim the string to account for double spaces
328  } else { // terminated by a > character
329  nAnglePos = strLine.find(">");
330  value = strLine.substr(0, nAnglePos);
331  }
332  trim(value); // always trim the string to account for double spaces
333  // remove the enclosing double quotes
334  value = value.substr(1, value.length() - 2);
335 
336  // Add the attribute as a key/value pair to the element's map
337  elem->addAttribute(key, value);
338  // cout << "key = " << key << " | value = " << value << endl ;
339  }
340 }
